```java
import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.io.IOException;
import java.util.Scanner;

public class ContarVocales {

    // Función contar vocales
    public static int contarVocales(String linea) {
        int contador = 0;
        for (char c : linea.toLowerCase().toCharArray()) {
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                contador++;
            }
        }
        return contador;
    }

    public static void main(String[] args) {
        try (Scanner scanner = new Scanner(System.in)) {
            // Nombre para el fichero
            System.out.print("Introduce el nombre del fichero: ");
            String nombreFichero = scanner.nextLine();

            File fichero = new File(nombreFichero);

            // Verifica si el fichero existe
            if (!fichero.exists()) {
                System.out.println("El fichero no existe.");
                return;
            }

            int totalVocales = 0;

            try (BufferedReader br = new BufferedReader(new FileReader(fichero))) {
                String linea;
                while ((linea = br.readLine()) != null) {
                    totalVocales += contarVocales(linea);
                }
            } catch (IOException e) {
                System.out.println("Error al leer el fichero: " + e.getMessage());
            }

            // Muestra resultado
            System.out.println("El número total de vocales en el fichero es: " + totalVocales);
        }
    }
}
```

``` java
import java.io.File;

public class ListDirectoryContents {

    public static void main(String[] args) {
        // Puedes cambiar esta ruta por la que desees listar
        String directoryPath = "/fichero.txt";  
        

        // Crear un objeto File para el directorio
        File directory = new File(directoryPath);

        // Comprobar si el directorio existe y es válido
        if (!directory.exists()) {
            System.out.println("El directorio no existe.");
            return;
        }

        if (!directory.isDirectory()) {
            System.out.println("La ruta proporcionada no es un directorio.");
            return;
        }

        // Obtener el contenido del directorio
        File[] filesList = directory.listFiles();

        if (filesList != null) {
            for (File file : filesList) {
                // Determinar si el elemento es un fichero o un directorio
                String type = file.isDirectory() ? "Directorio" : "Fichero";

                // Obtener el tamaño en kilobytes
                long sizeInKB = file.length() / 1024;

                // Imprimir el nombre, tipo y tamaño del archivo/directorio
                System.out.printf("%-30s %-10s %10d KB%n", file.getName(), type, sizeInKB);
            }
        } else {
            System.out.println("No se puede listar el contenido del directorio.");
        }
    }
}
```
